    // Wait for DOM to load
document.addEventListener('DOMContentLoaded', function () {

    // Get the save button
    const saveButton = document.getElementById('save_event');
    const eventForm = document.getElementById('event_form');

    // Add click event to save button
    saveButton.addEventListener('click', function () {

        // Check if form is valid (uses HTML5 required validation)
        if (!eventForm.checkValidity()) {
            // Show validation messages
            eventForm.reportValidity();
            return; // Stop if form is invalid
        }

        // Get form values
        let eventName = document.getElementById('event_name').value;
        let weekday = document.getElementById('event_weekday').value;
        let time = document.getElementById('event_time').value;
        let modality = document.getElementById('event_modality').value;
        let location = document.getElementById('event_location').value;
        let remoteUrl = document.getElementById('event_remote_url').value;
        let attendees = document.getElementById('event_attendees').value;

        // Log to console (for testing)
        console.log('Event saved:', {
            name: eventName,
            weekday: weekday,
            time: time,
            modality: modality,
            location: location,
            remoteUrl: remoteUrl,
            attendees: attendees
        });

        // Create event card and add to calendar
        addEventToCalendar(eventName, weekday, time);

        // Reset form
        eventForm.reset();

        // Close modal
        let modal = bootstrap.Modal.getInstance(document.getElementById('event_modal'));
        modal.hide();
    });
});

// Function to add event to calendar
function addEventToCalendar(eventName, weekday, time) {
    // Find the correct day column
    const dayColumn = document.getElementById(weekday);

    if (dayColumn) {
        // Create event card
        const eventCard = document.createElement('div');
        eventCard.className = 'card mb-2 bg-primary text-white';
        eventCard.innerHTML = `
            <div class="card-body p-2">
                <small class="fw-bold">${eventName}</small><br>
                <small>${time}</small>
            </div>
        `;

        // Add to day column
        dayColumn.appendChild(eventCard);
    }
}
